    {
         printf("%d\t", arr[i]);
    }
}   
 int ret_max( int arr[], int n)
 {
    int max=arr[0];
    for( int i=0; i<n; i++)
    {
        if(arr[i]>max)
           max=arr[i];
    }
    return max;
 }

int main()
{
    int arr[]={22,33,33,66,55,88,11,33,11,55};
    int fo=0, lo=0;
    int key=99;
    for(int i=0;i<10;i++)
    {
        if(arr[i]==key)
        {
            if((i>=fo) && (fo==0))
            {
                fo=i+1;
            }
            else
            {
                lo=i+1;
            }
        }
    }
    
    if(fo||lo)
    {
        printf("Search Successfull\n");
